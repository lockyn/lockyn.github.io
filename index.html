#include #include int main() { using namespace std; int n; cin >> n; stack p; stack help; while (n > 0) { int num; cin >> num; p.push(num); n--; } while (!p.empty()) { if (help.empty()) { help.push(p.top()); p.pop(); } int temp = p.top(); p.pop(); while (help.top() temp) help.push(temp); } while (!help.empty()) { p.push(help.top()); help.pop(); } while(!p.empty()) { cout << p.top()<<" "; p.pop(); } return 0; }
